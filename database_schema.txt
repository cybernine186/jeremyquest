CREATE TABLE users (
	id INT UNSIGNED NOT NULL AUTO_INCREMENT,
	username VARCHAR(30) NOT NULL,
	hash VARCHAR(100) NOT NULL,
	permission_handins TINYINT NOT NULL DEFAULT 0,
	permission_trades TINYINT NOT NULL DEFAULT 0,
	permission_looted TINYINT NOT NULL DEFAULT 0,
	permission_dropped TINYINT NOT NULL DEFAULT 0,
	permission_destroyed TINYINT NOT NULL DEFAULT 0,
	permission_inventory TINYINT NOT NULL DEFAULT 0,
	permission_logging TINYINT NOT NULL DEFAULT 0,
	permission_users TINYINT NOT NULL DEFAULT 0,
	permission_connections TINYINT NOT NULL DEFAULT 0,
	permission_copychar TINYINT NOT NULL DEFAULT 0,
	PRIMARY KEY (id)
);

CREATE TABLE cookiehashes (
	id INT UNSIGNED NOT NULL AUTO_INCREMENT,
	userid INT UNSIGNED NOT NULL,
	cookiehash VARCHAR(100) NOT NULL,
	created DATETIME NOT NULL,
	used DATETIME NOT NULL,
	PRIMARY KEY (id)
);

CREATE TABLE logs (
	id INT UNSIGNED NOT NULL AUTO_INCREMENT,
	time DATETIME NOT NULL,
	uid INT UNSIGNED NOT NULL,
	type TINYINT UNSIGNED NOT NULL,
	message VARCHAR(100),
	PRIMARY KEY (id)
);

CREATE TABLE connections (
	id INT UNSIGNED NOT NULL AUTO_INCREMENT,
	user INT UNSIGNED NOT NULL,
	name VARCHAR(30) NOT NULL,
	host VARCHAR(50) NOT NULL,
	dbase VARCHAR(20) NOT NULL,
	username VARCHAR(20) NOT NULL,
	password VARCHAR(30) NOT NULL,
	selected TINYINT NOT NULL DEFAULT 0,
    PRIMARY KEY (id)
);
